122void int_to_str(int32_t num, int base, char *str)
123{
124 char temp[35];
125 int i = 0, j = 0;
126 int is_negative = 0;
127
128 if (base < 2 || base > 16)
129 {
130 str[0] = '\0';
131 return;
132 }
133
134 if (num == 0)
135 {
136 str[j++] = '0';
137 str[j] = '\0';
138 return;
139 }
140
141 if (num < 0 && base == 10)
142 {
143 num = -num;
144 }
145
146 while (num != 0)
147 {
148 int rem = num % base;
149 temp[i++] = (rem > 9) ? (rem - 10) + 'A' : rem + '0';
150 num /= base;
151 }
152
153 if (is_negative)
154 {
155 temp[i++] = '-';
156 }
157
158 while (i > 0)
159 {
160 str[j++] = temp[--i];
161 }
162 str[j] = '\0';
163}

165int32_t parse_integer(const char *str)
166{
167 int result = 0;
168 int sign = 1;
169
170 while (*str == ' ')
171 {
172 str++;
173 }
174
175 if (*str == '-')
176 {
177 sign = -1;
178 str++;
179 }
180
181 if (*str < '0' || *str > '9')
182 {
183 return -1;
184 }
185
186 while (*str >= '0' && *str <= '9')
187 {
188 result = result * 10 + (*str - '0');
189 str++;
190 }
191
192 while (*str == ' ' || *str == '\n')
193 {
194 str++;
195 }
196
197 if (*str != '\0')
198 {
199 return -1;
200 }
201
202 return result * sign;
203}
